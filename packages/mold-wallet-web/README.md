```bash
npm install --save @owlmeans/regov-lib-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from '@owlmeans/regov-lib-react'
import '@owlmeans/regov-lib-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
